import torch
import torch.nn as nn
from torch.nn import Module, ModuleList, BatchNorm2d, MaxPool2d, BatchNorm1d, AdaptiveAvgPool2d

from brevitas.nn import QuantConv2d, QuantIdentity, QuantLinear
from brevitas.core.restrict_val import RestrictValueType

# 引入 BiasQuantSolver，這是專門處理 Bias 參數介面的
from brevitas.quant.solver import BiasQuantSolver

try:
    # 注意：這裡我們多引入了 CommonQuant，因為 Int8Bias 需要繼承它
    from brevitas_examples.bnn_pynq.models.common import CommonActQuant, CommonWeightQuant, CommonQuant
    from brevitas_examples.bnn_pynq.models.tensor_norm import TensorNorm
except ImportError:
    from .common import CommonActQuant, CommonWeightQuant, CommonQuant
    from .tensor_norm import TensorNorm

CNV_OUT_CH_POOL = [(64, False), (64, True), (128, False), (128, True), (256, False), (256, False)]
INTERMEDIATE_FC_FEATURES = [(256, 512), (512, 512)]
LAST_FC_IN_FEATURES = 512
KERNEL_SIZE = 3

# [修復重點] 定義 Int8Bias
# 1. 繼承 CommonQuant (取得基礎設定)
# 2. 繼承 BiasQuantSolver (取得 Bias 專用的介面)
# 3. 補上 scaling_const (解決 DependencyError)
class Int8Bias(CommonQuant, BiasQuantSolver):
    bit_width = 8
    requires_input_scale = False
    requires_weight_scale = False
    scaling_const = 1.0  # <--- 關鍵修復：補上這個缺失的屬性

# =============================================================================
#  Adapter 定義區
# =============================================================================

class QuantAdapter(Module):
    def __init__(self, in_channels, out_channels, reduction=4, bit_width=1, input_bit_width=1, rc_bit_width=8):
        super(QuantAdapter, self).__init__()
        reduced_channels = max(1, in_channels // reduction)
        
        # 為了避開 FINN 的 XNOR 檢查，如果輸入大於 1-bit，第一層權重強制 8-bit
        if input_bit_width > 1:
            down_weight_bit = 8  
        else:
            down_weight_bit = bit_width 

        # 設定 RC (Bias) 的量化器
        bias_quantizer = Int8Bias if rc_bit_width == 8 else None

        self.down = QuantConv2d(
            in_channels=in_channels, 
            out_channels=reduced_channels, 
            kernel_size=1,            
            bias=True,                # 開啟 Bias 作為 RC
            bias_quant=bias_quantizer, # 使用正確的 Bias 量化器
            weight_quant=CommonWeightQuant, 
            weight_bit_width=down_weight_bit 
        )
        
        self.act = QuantIdentity(act_quant=CommonActQuant, bit_width=bit_width)
        
        self.up = QuantConv2d(
            in_channels=reduced_channels, 
            out_channels=out_channels, 
            kernel_size=1, 
            bias=False, 
            weight_quant=CommonWeightQuant, 
            weight_bit_width=bit_width
        )

    def forward(self, x):
        x = self.down(x)
        x = self.act(x)
        x = self.up(x)
        return x

class MultiBranchAdapter(Module):
    def __init__(self, in_channels, out_channels, num_branches=1, bit_width=1, rc_bit_width=8, use_rc=True, input_bit_width=1):
        super(MultiBranchAdapter, self).__init__()
        self.use_rc = use_rc
        self.branches = ModuleList()
        self.alphas = nn.ParameterList()
        
        for _ in range(num_branches):
            self.branches.append(QuantAdapter(
                in_channels, 
                out_channels, 
                bit_width=bit_width, 
                input_bit_width=input_bit_width,
                rc_bit_width=rc_bit_width 
            ))
            self.alphas.append(nn.Parameter(torch.tensor(1.0)))

    def forward(self, x):
        out = 0
        for i in range(len(self.branches)):
            branch_out = self.branches[i](x)
            
            if self.training:
                alpha_val = self.alphas[i]
            else:
                alpha_val = float(self.alphas[i]) 
            
            out = out + branch_out * alpha_val
        return out

# =============================================================================
#  主模型 CNV
# =============================================================================

class CNV(Module):
    def __init__(self, num_classes, weight_bit_width, act_bit_width, in_bit_width, in_ch, 
                 adapter_config=None):
        super(CNV, self).__init__()
        self.use_adapter = False
        self.adapter_config = adapter_config
        self.return_features = False 

        self.conv_features = ModuleList()
        self.linear_features = ModuleList()
        self.adapters = ModuleList()

        current_input_bit_width = in_bit_width

        self.conv_features.append(QuantIdentity(
            act_quant=CommonActQuant,
            bit_width=in_bit_width,
            min_val=-1.0, max_val=1.0 - 2.0 ** (-7),
            narrow_range=False,
            restrict_scaling_type=RestrictValueType.POWER_OF_TWO))

        current_in_ch = in_ch

        for out_ch, is_pool_enabled in CNV_OUT_CH_POOL:
            # 1. 卷積
            self.conv_features.append(QuantConv2d(
                kernel_size=KERNEL_SIZE, in_channels=current_in_ch, out_channels=out_ch,
                padding=1, 
                bias=False, weight_quant=CommonWeightQuant, weight_bit_width=weight_bit_width))
            
            # 2. 插入 Adapter
            if adapter_config:
                self.use_adapter = True
                self.adapters.append(MultiBranchAdapter(
                    in_channels=current_in_ch,
                    out_channels=out_ch,
                    num_branches=adapter_config['num_branches'],
                    bit_width=adapter_config['bit_width'],
                    rc_bit_width=adapter_config['rc_bit_width'],
                    use_rc=adapter_config.get('use_rc', True),
                    input_bit_width=current_input_bit_width 
                ))
            else:
                self.adapters.append(nn.Identity())

            current_in_ch = out_ch
            
            # 3. BN 和 Activation
            self.conv_features.append(BatchNorm2d(current_in_ch, eps=1e-4))
            self.conv_features.append(QuantIdentity(act_quant=CommonActQuant, bit_width=act_bit_width))
            
            # 4. Pooling
            if is_pool_enabled:
                self.conv_features.append(MaxPool2d(kernel_size=2))
            
            current_input_bit_width = act_bit_width

        self.avg_pool = AdaptiveAvgPool2d((1, 1))

        for in_features, out_features in INTERMEDIATE_FC_FEATURES:
            self.linear_features.append(QuantLinear(
                in_features=in_features, out_features=out_features,
                bias=False, weight_quant=CommonWeightQuant, weight_bit_width=weight_bit_width))
            self.linear_features.append(BatchNorm1d(out_features, eps=1e-4))
            self.linear_features.append(QuantIdentity(act_quant=CommonActQuant, bit_width=act_bit_width))

        self.linear_features.append(QuantLinear(
            in_features=LAST_FC_IN_FEATURES, out_features=num_classes,
            bias=False, weight_quant=CommonWeightQuant, weight_bit_width=weight_bit_width))
        self.linear_features.append(TensorNorm())

        for m in self.modules():
            if isinstance(m, QuantConv2d) or isinstance(m, QuantLinear):
                torch.nn.init.uniform_(m.weight.data, -1, 1)

    def clip_weights(self, min_val, max_val):
        for mod in self.conv_features:
            if isinstance(mod, QuantConv2d):
                mod.weight.data.clamp_(min_val, max_val)
        for mod in self.linear_features:
            if isinstance(mod, QuantLinear):
                mod.weight.data.clamp_(min_val, max_val)
        # Adapter 也要 Clip
        for mod in self.adapters:
            if isinstance(mod, MultiBranchAdapter):
                for branch in mod.branches:
                    branch.down.weight.data.clamp_(min_val, max_val)
                    branch.up.weight.data.clamp_(min_val, max_val)

    def forward(self, x):
        self.features = [] 
        x = 2.0 * x - torch.tensor([1.0], device=x.device)
        
        if not self.use_adapter:
            for mod in self.conv_features: x = mod(x)
            x = self.avg_pool(x)
        else:
            adapter_idx = 0
            x = self.conv_features[0](x) 
            current_layer_idx = 1
            
            for out_ch, is_pool_enabled in CNV_OUT_CH_POOL:
                input_tensor = x 
                x_conv = self.conv_features[current_layer_idx](x)
                current_layer_idx += 1
                
                # 執行 Adapter
                if isinstance(self.adapters[adapter_idx], MultiBranchAdapter):
                    x_adapter = self.adapters[adapter_idx](input_tensor)
                    x = x_conv + x_adapter
                    if self.return_features: self.features.append(x)
                else:
                    x = x_conv
                
                adapter_idx += 1
                x = self.conv_features[current_layer_idx](x); current_layer_idx += 1 
                x = self.conv_features[current_layer_idx](x); current_layer_idx += 1 
                if is_pool_enabled:
                    x = self.conv_features[current_layer_idx](x); current_layer_idx += 1 
            
            x = self.avg_pool(x)

        x = x.view(x.shape[0], -1)
        for mod in self.linear_features:
            x = mod(x)
        
        if self.return_features: return x, self.features
        return x

def cnv(cfg):
    weight_bit_width = cfg.getint('QUANT', 'WEIGHT_BIT_WIDTH')
    act_bit_width = cfg.getint('QUANT', 'ACT_BIT_WIDTH')
    in_bit_width = cfg.getint('QUANT', 'IN_BIT_WIDTH')
    num_classes = cfg.getint('MODEL', 'NUM_CLASSES')
    in_channels = cfg.getint('MODEL', 'IN_CHANNELS')
    
    adapter_config = None
    if cfg.has_section('ADAPTER'):
        adapter_config = {
            'num_branches': cfg.getint('ADAPTER', 'NUM_BRANCHES'),
            'bit_width': cfg.getint('ADAPTER', 'BIT_WIDTH'),
            'rc_bit_width': cfg.getint('ADAPTER', 'RC_BIT_WIDTH'),
            'use_rc': cfg.getboolean('ADAPTER', 'USE_RC') if cfg.has_option('ADAPTER', 'USE_RC') else True
        }

    net = CNV(
        weight_bit_width=weight_bit_width,
        act_bit_width=act_bit_width,
        in_bit_width=in_bit_width,
        num_classes=num_classes,
        in_ch=in_channels,
        adapter_config=adapter_config)
    return net
